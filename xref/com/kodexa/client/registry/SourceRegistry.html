8   import java.io.InputStream;
9   import java.util.HashMap;
10  import java.util.Map;
11  
12  /**
13   * The registry of source types, used in Kodexa to allow us to
14   * separate access to sources from the steps and pipeline
15   */
16  public class SourceRegistry {
17  
18      private static SourceRegistrySourceRegistry">SourceRegistry INTERNAL_REGISTRY = new SourceRegistry();
19  
20      static {
21          INTERNAL_REGISTRY.addConnector(new FolderConnector());
22      }
23  
24      private Map<String, Connector> connectors = new HashMap<>();
25  
26      public static SourceRegistry getInstance() {
27          return INTERNAL_REGISTRY;
28      }
29  
30      public void addConnector(Connector connector) {
31          connectors.put(connector.getName(), connector);
32      }
33  
34      public InputStream getSource(Document document) {
35          String CONNECTOR_KEY = "connector";
36          if (document.getMetadata().containsKey(CONNECTOR_KEY)) {
37              String connector = String.valueOf(document.getMetadata().get(CONNECTOR_KEY));
38              if (connectors.containsKey(connector)) {
39                  return connectors.get(connector).getSource(document);
40              } else {
41                  throw new KodexaException("Unable to find connector [" + connector + "]");
42              }
43          } else {
44              throw new KodexaException("Document does not have connector metadata");
45          }
46      }
47  }
