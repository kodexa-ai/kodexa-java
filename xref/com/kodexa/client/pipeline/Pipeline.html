1   package com.kodexa.client.pipeline;
2   
3   import com.kodexa.client.Document;
4   import com.kodexa.client.connectors.Connector;
5   import com.kodexa.client.sink.Sink;
6   import com.kodexa.client.steps.PipelineStep;
7   import lombok.extern.slf4j.Slf4j;
8   
9   import java.util.ArrayList;
10  import java.util.List;
11  
12  /**
13   * A Pipeline allows you to put together steps from a connector to a sink
14   * to enable re-use and encapsulation of steps.
15   */
16  @Slf4j
17  public class Pipeline {
18  
19      protected final Connector connector;
20      private final PipelineContext context;
21      private Sink sink;
22      private List<PipelineStep> steps = new ArrayList<>();
23  
24      public Pipeline(Connector connector) {
25          this.connector = connector;
26          this.context = new PipelineContext();
27      }
28  
29      public Pipeline(Document document) {
30          this.connector = new DocumentConnector(document);
31          this.context = new PipelineContext();
32      }
33  
34      public void addStep(PipelineStep step) {
35          steps.add(step);
36      }
37  
38      public void setSink(Sink sink) {
39          this.sink = sink;
40      }
41  
42      public PipelineContext run() {
43  
44          log.info("Starting pipeline");
45  
46          connector.forEachRemaining(document -> {
47              for (PipelineStep step : steps) {
48                  log.info("Starting step " + step.getName());
49                  long startTime = System.currentTimeMillis();
50                  document = step.process(document, context);
51                  long endTime = System.currentTimeMillis();
52                  log.info("Step processed in " + (endTime - startTime) + " ms");
53              }
54  
55              if (sink != null) {
56                  log.info("Writing to sink " + sink.getName());
57                  sink.sink(document);
58              }
59          });
60  
61          log.info("Pipeline completed");
62          return context;
63  
64      }
65  
66      public static Pipeline fromText(String text) {
67          return new Pipeline(Document.fromText(text));
68      }
69  
70      public static Pipeline fromUrl(String url) {
71          return new Pipeline(Document.fromUrl(url));
72      }
73  
74  }
