4   import com.kodexa.client.connectors.Connector;
5   import com.kodexa.client.connectors.FolderConnector;
6   import com.kodexa.client.connectors.InputStreamConnector;
7   import com.kodexa.client.remote.RemoteAction;
8   import com.kodexa.client.sink.Sink;
9   import com.kodexa.client.steps.PipelineStep;
10  import lombok.extern.slf4j.Slf4j;
11  
12  import java.io.InputStream;
13  import java.util.ArrayList;
14  import java.util.List;
15  
16  /**
17   * A Pipeline allows you to put together steps from a connector to a sink
18   * to enable re-use and encapsulation of steps.
19   */
20  @Slf4j
21  public class Pipeline {
22  
23      protected final Connector connector;
24      private final PipelineContext context;
25      private Sink sink;
26      private List<PipelineStepWrapper> steps = new ArrayList<>();
27      protected List<PipelineParameter> parameters = new ArrayList<>();
28  
29      public Pipeline(Connector connector) {
30          this.connector = connector;
31          this.context = new PipelineContext();
32      }
33  
34      public Pipeline(Document document) {
35          this.connector = new DocumentConnector(document);
36          this.context = new PipelineContext();
37      }
38  
39      public Pipeline addStep(Class stepClass, Options options) {
40          steps.add(new PipelineStepWrapper(new ClassBasedStep(stepClass), options));
41          return this;
42      }
43  
44      public Pipeline addStep(String actionSlug, Options options) {
45          steps.add(new PipelineStepWrapper(new RemoteAction(actionSlug), options));
46          return this;
47      }
48  
49      public Pipeline addStep(PipelineStep step) {
50          steps.add(new PipelineStepWrapper(step, new Options()));
51          return this;
52      }
53  
54      public Pipeline setSink(Sink sink) {
55          this.sink = sink;
56          return this;
57      }
58  
59      public Pipeline parameters(List<PipelineParameter> parameters) {
60          this.parameters = parameters;
61          return this;
62      }
63  
64      public PipelineContext run() {
65  
66          log.info("Starting pipeline");
67  
68          this.context.setParameters(this.parameters);
69  
70          connector.forEachRemaining(document -> {
71              for (PipelineStepWrapper step : steps) {
72                  long startTime = System.currentTimeMillis();
73                  document = step.process(document, context);
74                  long endTime = System.currentTimeMillis();
75                  log.info("Step processed in " + (endTime - startTime) + " ms");
76              }
77  
78              if (sink != null) {
79                  log.info("Writing to sink " + sink.getName());
80                  sink.sink(document);
81              }
82  
83              context.setOutputDocument(document);
84          });
85  
86          log.info("Pipeline completed");
87          return context;
88  
89      }
90  
91      /**
92       * Create a pipeline with a single document as input containing the content from this text
93       *
94       * @param text The text to use as the content of the document
95       * @return an instance of the pipeline
96       */
97      public static Pipeline fromText(String text) {
98          return new Pipeline(Document.fromText(text));
99      }
100 
101     public static Pipeline fromUrl(String url) {
102         return new Pipeline(Document.fromUrl(url));
103     }
104 
105     public static Pipeline fromInputStream(InputStream inputStream) {
106         return new Pipeline(new InputStreamConnector(inputStream));
107     }
108 
109     public static Pipeline fromFolder(String folderPath, String filenameFilter, boolean recursive) {
110         return new Pipeline(new FolderConnector(folderPath, filenameFilter, recursive));
111     }
112 
113     public Pipeline addParameter(String name, String value) {
114         parameters.add(new PipelineParameter(name, value));
115         return this;
116     }
117 }
