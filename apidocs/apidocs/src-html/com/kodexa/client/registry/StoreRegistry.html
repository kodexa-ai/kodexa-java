001package com.kodexa.client.registry;
002
003import com.kodexa.client.KodexaException;
004import com.kodexa.client.store.DataStore;
005import com.kodexa.client.store.DictionaryStore;
006import com.kodexa.client.store.TableStore;
007
008import java.util.Map;
009
010
011// TODO need to make sure this is plugin/extensible
012public class StoreRegistry {
013
014    private static StoreRegistry INTERNAL_REGISTRY = new StoreRegistry();
015
016    public static StoreRegistry getInstance() {
017        return INTERNAL_REGISTRY;
018    }
019
020    public DataStore deserialize(String type, Map<String, Object> data) {
021
022        switch (type) {
023            case "table": {
024                return new TableStore(data);
025            }
026            case "dictionary": {
027                return new DictionaryStore(data);
028            }
029        }
030
031        throw new KodexaException("Unknown store");
032    }
033
034}
