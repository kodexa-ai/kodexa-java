001package com.kodexa.client.registry;
002
003import com.kodexa.client.Document;
004import com.kodexa.client.KodexaException;
005import com.kodexa.client.connectors.Connector;
006import com.kodexa.client.connectors.FolderConnector;
007import com.kodexa.client.connectors.InputStreamConnector;
008
009import java.io.InputStream;
010import java.util.HashMap;
011import java.util.Map;
012
013/**
014 * The registry of source types, used in Kodexa to allow us to
015 * separate access to sources from the steps and pipeline
016 */
017public class SourceRegistry {
018
019    private static SourceRegistry INTERNAL_REGISTRY = new SourceRegistry();
020
021    static {
022        INTERNAL_REGISTRY.addConnector(new FolderConnector());
023        INTERNAL_REGISTRY.addConnector(new InputStreamConnector());
024    }
025
026    private Map<String, Connector> connectors = new HashMap<>();
027
028    public static SourceRegistry getInstance() {
029        return INTERNAL_REGISTRY;
030    }
031
032    public void addConnector(Connector connector) {
033        connectors.put(connector.getName(), connector);
034    }
035
036    public InputStream getSource(Document document) {
037        String CONNECTOR_KEY = "connector";
038        if (document.getMetadata().containsKey(CONNECTOR_KEY)) {
039            String connector = String.valueOf(document.getMetadata().get(CONNECTOR_KEY));
040            if (connectors.containsKey(connector)) {
041                return connectors.get(connector).getSource(document);
042            } else {
043                throw new KodexaException("Unable to find connector [" + connector + "]");
044            }
045        } else {
046            throw new KodexaException("Document does not have connector metadata");
047        }
048    }
049}
