102                post.addHeader("x-access-token", KodexaCloud.getAccessToken());
103                HttpResponse response = client.execute(post);
104                if (response.getStatusLine().getStatusCode() != 200) {
105                    throw new KodexaException("Unable to create a session on Kodexa, check your access token and URL [" + response.getStatusLine().getStatusCode() + "]");
106                }
107                String responseJson = IOUtils.toString(response.getEntity().getContent(), StandardCharsets.UTF_8);
108                execution = jsonOm.readValue(responseJson, CloudExecution.class);
109                if (!execution.getStatus().equals(status)) {
110                    log.info("Execution Status changed from " + status + " => " + execution.getStatus());
111                    status = execution.getStatus();
112                }
113            } catch (IOException | InterruptedException e) {
114                throw new KodexaException("Unable to connect to Kodexa", e);
115            }
116        }
117        if (execution.getStatus().equals("FAILED")) {
118            CloudExceptionDetail exceptionDetail = execution.getExceptionDetail();
119            log.error("Failed: " + exceptionDetail.getMessage());
120            if (exceptionDetail.getErrorType() != null) log.error("Exception Type: " + exceptionDetail.getErrorType());
121            log.error("More information is available:\n\n" + exceptionDetail.getHelp() + "\n");
122            throw new KodexaException("Failed:" + exceptionDetail.getMessage() + "]");
123        } else {
124            log.info("Execution completed with " + execution.getStores().size() + " stores and " + execution.getContentObjects().size() + " content objects");
125        }
126        return execution;
