010/**
011 * A Pipeline allows you to put together steps from a connector to a sink
012 * to enable re-use and encapsulation of steps.
013 */
014public class Pipeline {
015    @java.lang.SuppressWarnings("all")
016    private static final org.slf4j.Logger log = org.slf4j.LoggerFactory.getLogger(Pipeline.class);
017    private final Connector connector;
018    private final PipelineContext context;
019    private Sink sink;
020    private List<PipelineStep> steps = new ArrayList<>();
021
022    public Pipeline(Connector connector) {
023        this.connector = connector;
024        this.context = new PipelineContext();
025    }
026
027    public void addStep(PipelineStep step) {
028        steps.add(step);
029    }
030
031    public void setSink(Sink sink) {
032        this.sink = sink;
033    }
034
035    public PipelineContext run() {
036        log.info("Starting pipeline");
037        connector.forEachRemaining(document -> {
038            for (PipelineStep step : steps) {
039                log.info("Starting step " + step.getName());
040                long startTime = System.currentTimeMillis();
041                document = step.process(document, context);
042                long endTime = System.currentTimeMillis();
043                log.info("Step processed in " + (endTime - startTime) + " ms");
044            }
045            if (sink != null) {
046                log.info("Writing to sink " + sink.getName());
047                sink.sink(document);
048            }
049        });
050        log.info("Pipeline completed");
051        return context;
052    }
