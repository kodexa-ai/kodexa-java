001package com.kodexa.client.pipeline;
002
003import com.kodexa.client.Document;
004import com.kodexa.client.KodexaException;
005import com.kodexa.client.steps.PipelineStep;
006
007public class ClassBasedStep implements PipelineStep, OptionDrivenStep {
008
009    private final Class stepClass;
010    private Options options;
011
012    public ClassBasedStep(Class stepClass) {
013        this.stepClass = stepClass;
014    }
015
016    @Override
017    public Document process(Document document, PipelineContext context) {
018
019        Object classInstance = null;
020        try {
021            classInstance = this.stepClass.newInstance();
022            if (classInstance instanceof OptionDrivenStep) {
023                ((OptionDrivenStep) classInstance).setOptions(this.options);
024            }
025            if (classInstance instanceof PipelineStep) {
026                return ((PipelineStep) classInstance).process(document, context);
027            } else {
028                throw new KodexaException("Class is not an instance of PipelineStep?");
029            }
030        } catch (Exception e) {
031            throw new KodexaException("Unable to create instance of class for step", e);
032        }
033
034
035    }
036
037    @Override
038    public String getName() {
039        return "Step for class " + stepClass.getName();
040    }
041
042    @Override
043    public void setOptions(Options options) {
044        this.options = options;
045    }
046}
