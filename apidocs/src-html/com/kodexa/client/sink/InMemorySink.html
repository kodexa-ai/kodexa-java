001package com.kodexa.client.sink;
002
003import com.kodexa.client.Document;
004
005import java.util.ArrayList;
006import java.util.List;
007
008/**
009 * An in-memory sink for capturing documents from a pipeline
010 */
011public class InMemorySink implements Sink {
012
013    private List<Document> documents = new ArrayList<>();
014
015    public Document getDocument(int i) {
016        return documents.get(i);
017    }
018
019    public int getCount() {
020        return documents.size();
021    }
022
023    @Override
024    public void sink(Document document) {
025        documents.add(document);
026    }
027
028    @Override
029    public String getName() {
030        return "In-memory";
031    }
032
033}
