082    protected CloudExecution waitForExecution(CloudSession session, CloudExecution execution) {
083        String status = execution.getStatus();
084        while (execution.getStatus().equals("PENDING") || execution.getStatus().equals("RUNNING")) {
085            try (CloseableHttpClient client = HttpClientBuilder.create().setDefaultRequestConfig(getRequestConfig()).build()) {
086                Thread.sleep(1000);
087                String url = KodexaPlatform.getUrl() + "/api/sessions/" + session.getId() + "/executions/" + execution.getId();
088                HttpGet post = new HttpGet(url);
089                post.addHeader("x-access-token", KodexaPlatform.getAccessToken());
090                HttpResponse response = client.execute(post);
091                if (response.getStatusLine().getStatusCode() != 200) {
092                    throw new KodexaException("Unable to create a session on Kodexa, check your access token and URL [" + response.getStatusLine().getStatusCode() + "]");
093                }
094                String responseJson = IOUtils.toString(response.getEntity().getContent(), StandardCharsets.UTF_8);
095                execution = jsonOm.readValue(responseJson, CloudExecution.class);
096                if (!execution.getStatus().equals(status)) {
097                    log.info("Execution Status changed from " + status + " => " + execution.getStatus());
098                    status = execution.getStatus();
099                }
100            } catch (IOException | InterruptedException e) {
101                throw new KodexaException("Unable to connect to Kodexa", e);
102            }
103        }
104        if (execution.getStatus().equals("FAILED")) {
105            CloudExceptionDetail exceptionDetail = execution.getExceptionDetail();
106            log.error("Failed: " + exceptionDetail.getMessage());
107            if (exceptionDetail.getErrorType() != null) log.error("Exception Type: " + exceptionDetail.getErrorType());
108            log.error("More information is available:\n\n" + exceptionDetail.getHelp() + "\n");
109            throw new KodexaException("Failed:" + exceptionDetail.getMessage() + "]");
110        } else {
111            log.info("Execution completed with " + execution.getStores().size() + " stores and " + execution.getContentObjects().size() + " content objects");
112        }
113        return execution;
114    }
