001package com.kodexa.client;
002
003import com.fasterxml.jackson.annotation.JsonInclude;
004import com.fasterxml.jackson.core.type.TypeReference;
005import com.fasterxml.jackson.databind.ObjectMapper;
006import org.apache.commons.io.IOUtils;
007import org.jdbi.v3.core.Handle;
008import org.jdbi.v3.core.Jdbi;
009import org.msgpack.jackson.dataformat.MessagePackFactory;
010
011import java.io.File;
012import java.io.FileOutputStream;
013import java.io.IOException;
014import java.io.InputStream;
015import java.nio.file.Files;
016import java.nio.file.Path;
017import java.util.ArrayList;
018import java.util.HashMap;
019import java.util.List;
020import java.util.Map;
021import java.util.stream.Collectors;
022
023/**
024 * A Persistence Layer that works with the document to allow it to be interacted with
025 */
026public class SqlitePersistenceLayer {
027
028    private final static ObjectMapper OBJECT_MAPPER_MSGPACK;
029
030    Map<Integer, String> nodeTypes;
031
032    Map<Integer, String> featureTypeNames;
033
034    static {
035        OBJECT_MAPPER_MSGPACK = new ObjectMapper(new MessagePackFactory());
036        OBJECT_MAPPER_MSGPACK.setSerializationInclusion(JsonInclude.Include.NON_NULL);
037    }
038
039    private String dbPath;
040    private Jdbi jdbi;
041
042    public SqlitePersistenceLayer() {
043        this.dbPath = ":memory:";
044        this.initializeLayer();
045    }
046
047    public SqlitePersistenceLayer(InputStream kddbInputStream) {
048        final File tempFile;
049        try {
050            tempFile = File.createTempFile("kodexa", "kddb");
051            tempFile.deleteOnExit();
052            try (FileOutputStream out = new FileOutputStream(tempFile)) {
053                IOUtils.copy(kddbInputStream, out);
054            }
055            this.dbPath = tempFile.getAbsolutePath();
056
057            this.initializeLayer();
058        } catch (IOException e) {
059            throw new KodexaException("Unable to create persistence layer for KDDB object", e);
060        }
061    }
062
063    public SqlitePersistenceLayer(File kddbFile) {
064        this.dbPath = kddbFile.getAbsolutePath();
065        this.initializeLayer();
066    }
067
068    private void initializeLayer() {
069        try {
070            Class.forName("org.sqlite.JDBC");
071            jdbi = Jdbi.create("jdbc:sqlite:" + dbPath);
072        } catch (ClassNotFoundException e) {
073            throw new KodexaException("Unable to create persistence layer for KDDB object", e);
074        }
075    }
076
077    protected void loadDocument(Document document) {
078        // This method will update the document to match the contents
079        // of the KDDB database
080        jdbi.withHandle(handle -> {
081
082            // We get all the metadata back
083            try {
084                byte[] metadataPack = (byte[]) handle.createQuery("SELECT * FROM metadata")
085                        .mapToMap()
086                        .first().get("metadata");
087
088                Document baseDocument = OBJECT_MAPPER_MSGPACK.readValue(metadataPack, Document.class);
089                document.setSource(baseDocument.getSource());
090                document.setClasses(baseDocument.getClasses());
091                document.setLabels(baseDocument.getLabels());
092                document.setMetadata(baseDocument.getMetadata());
093                document.setUuid(baseDocument.getUuid());
094
095
096                // Lets get all the node types and feature type/name combinations
097                nodeTypes =
098                        handle.createQuery("SELECT id, name FROM n_type")
099                                .mapToMap()
100                                .collect(Collectors.toMap(x -> Integer.valueOf(String.valueOf(x.get("id"))), x -> String.valueOf(x.get("name"))));
101
102                featureTypeNames =
103                        handle.createQuery("SELECT id, name FROM f_type")
104                                .mapToMap()
105                                .collect(Collectors.toMap(x -> Integer.valueOf(String.valueOf(x.get("id"))), x -> String.valueOf(x.get("name"))));
106            } catch (IOException e) {
107                throw new KodexaException("Unable to unpack metadata", e);
108            }
109
110            List<Map<String, Object>> contentNodes =
111                    handle.createQuery("SELECT id, nt, pid FROM cn where pid is null")
112                            .mapToMap()
113                            .list();
114            for (Map<String, Object> contentNode : contentNodes) {
115                document.setContentNode(buildNode(contentNode, handle));
116            }
117
118            return contentNodes;
119        });
120
121    }
122
123    private ContentNode buildNode(Map<String, Object> contentNodeValues, Handle handle) {
124        ContentNode contentNode = new ContentNode();
125        contentNode.setUuid(String.valueOf(contentNodeValues.get("id")));
126
127        List<Map<String, Object>> features =
128                handle.createQuery("SELECT id, f_type, fvalue_id FROM f where cn_id is :nodeId").bind("nodeId", contentNode.getUuid())
129                        .mapToMap()
130                        .list();
131
132        for (Map<String, Object> feature : features) {
133            Map<String, Object> featureValue =
134                    handle.createQuery("SELECT binary_value, single FROM f_value where id is :fvalue_id").bind("fvalue_id", feature.get("fvalue_id"))
135                            .mapToMap()
136                            .first();
137            ContentFeature contentFeature = new ContentFeature();
138            contentFeature.setFeatureType(featureTypeNames.get(feature.get("f_type")).split(":")[0]);
139            contentFeature.setName(featureTypeNames.get(feature.get("f_type")).split(":")[1]);
140            contentFeature.setSingle(Integer.valueOf(1) == featureValue.get("single"));
141            TypeReference<ArrayList<Object>> typeRef
142                    = new TypeReference<>() {
143            };
144            try {
145                contentFeature.setValue(OBJECT_MAPPER_MSGPACK.readValue((byte[]) featureValue.get("binary_value"), typeRef));
146            } catch (IOException e) {
147                throw new KodexaException("Unable to unpack value for feature", e);
148            }
149        }
150
151        List<Map<String, Object>> childNodes =
152                handle.createQuery("SELECT id, nt, pid FROM cn where pid is :nodeId").bind("nodeId", contentNode.getUuid())
153                        .mapToMap()
154                        .list();
155
156        for (Map<String, Object> childNode : childNodes) {
157            contentNode.getChildren().add(buildNode(childNode, handle));
158        }
159
160        return contentNode;
161    }
162
163    public byte[] toBytes() throws IOException {
164        return Files.readAllBytes(Path.of(dbPath));
165    }
166}
