001package com.kodexa.client;
002
003import com.fasterxml.jackson.annotation.JsonInclude;
004import com.fasterxml.jackson.core.type.TypeReference;
005import com.fasterxml.jackson.databind.ObjectMapper;
006import org.apache.commons.io.IOUtils;
007import org.jdbi.v3.core.Handle;
008import org.jdbi.v3.core.Jdbi;
009import org.msgpack.jackson.dataformat.MessagePackFactory;
010
011import java.io.File;
012import java.io.FileOutputStream;
013import java.io.IOException;
014import java.io.InputStream;
015import java.nio.file.Files;
016import java.nio.file.Path;
017import java.util.ArrayList;
018import java.util.HashMap;
019import java.util.List;
020import java.util.Map;
021import java.util.stream.Collectors;
022
023/**
024 * A Persistence Layer that works with the document to allow it to be interacted with
025 */
026public class SqlitePersistenceLayer {
027
028    private final static ObjectMapper OBJECT_MAPPER_MSGPACK;
029
030    Map<Integer, String> nodeTypes;
031
032    Map<Integer, String> featureTypeNames;
033
034    static {
035        OBJECT_MAPPER_MSGPACK = new ObjectMapper(new MessagePackFactory());
036        OBJECT_MAPPER_MSGPACK.setSerializationInclusion(JsonInclude.Include.NON_NULL);
037    }
038
039    private String dbPath;
040    private Jdbi jdbi;
041    private boolean tempFile = false;
042
043    public SqlitePersistenceLayer() {
044        File file = null;
045        try {
046            file = File.createTempFile("kdx", "kddb");
047            this.dbPath = file.getAbsolutePath();
048            this.tempFile = true;
049            file.deleteOnExit();
050            this.initializeLayer();
051        } catch (IOException e) {
052            throw new KodexaException("Unable to initialize the temp file for KDDB", e);
053        }
054    }
055
056    public void close() {
057        if (tempFile) {
058            try {
059                Files.delete(Path.of(this.dbPath));
060            } catch (IOException e) {
061                throw new KodexaException("Unable to delete temp file", e);
062            }
063        }
064    }
065
066    public SqlitePersistenceLayer(InputStream kddbInputStream) {
067        final File tempFile;
068        try {
069            tempFile = File.createTempFile("kodexa", "kddb");
070            tempFile.deleteOnExit();
071            try (FileOutputStream out = new FileOutputStream(tempFile)) {
072                IOUtils.copy(kddbInputStream, out);
073            }
074            this.dbPath = tempFile.getAbsolutePath();
075
076            this.initializeLayer();
077        } catch (IOException e) {
078            throw new KodexaException("Unable to create persistence layer for KDDB object", e);
079        }
080    }
081
082    public SqlitePersistenceLayer(File kddbFile) {
083        this.dbPath = kddbFile.getAbsolutePath();
084        this.initializeLayer();
085    }
086
087    private void initializeLayer() {
088        try {
089            Class.forName("org.sqlite.JDBC");
090            jdbi = Jdbi.create("jdbc:sqlite:" + dbPath);
091        } catch (ClassNotFoundException e) {
092            throw new KodexaException("Unable to create persistence layer for KDDB object", e);
093        }
094    }
095
096    protected void loadDocument(Document document) {
097        // This method will update the document to match the contents
098        // of the KDDB database
099        jdbi.withHandle(handle -> {
100
101            // We get all the metadata back
102            try {
103                byte[] metadataPack = (byte[]) handle.createQuery("SELECT * FROM metadata")
104                        .mapToMap()
105                        .first().get("metadata");
106
107                Document baseDocument = OBJECT_MAPPER_MSGPACK.readValue(metadataPack, Document.class);
108                document.setSource(baseDocument.getSource());
109                document.setClasses(baseDocument.getClasses());
110                document.setLabels(baseDocument.getLabels());
111                document.setMetadata(baseDocument.getMetadata());
112                document.setUuid(baseDocument.getUuid());
113
114
115                // Lets get all the node types and feature type/name combinations
116                nodeTypes =
117                        handle.createQuery("SELECT id, name FROM n_type")
118                                .mapToMap()
119                                .collect(Collectors.toMap(x -> Integer.valueOf(String.valueOf(x.get("id"))), x -> String.valueOf(x.get("name"))));
120
121                featureTypeNames =
122                        handle.createQuery("SELECT id, name FROM f_type")
123                                .mapToMap()
124                                .collect(Collectors.toMap(x -> Integer.valueOf(String.valueOf(x.get("id"))), x -> String.valueOf(x.get("name"))));
125            } catch (IOException e) {
126                throw new KodexaException("Unable to unpack metadata", e);
127            }
128
129            List<Map<String, Object>> contentNodes =
130                    handle.createQuery("SELECT id, nt, pid FROM cn where pid is null")
131                            .mapToMap()
132                            .list();
133            for (Map<String, Object> contentNode : contentNodes) {
134                document.setContentNode(buildNode(contentNode, handle));
135            }
136
137            return contentNodes;
138        });
139
140    }
141
142    private ContentNode buildNode(Map<String, Object> contentNodeValues, Handle handle) {
143        ContentNode contentNode = new ContentNode();
144        contentNode.setUuid(String.valueOf(contentNodeValues.get("id")));
145
146        List<Map<String, Object>> features =
147                handle.createQuery("SELECT id, f_type, fvalue_id FROM f where cn_id is :nodeId").bind("nodeId", contentNode.getUuid())
148                        .mapToMap()
149                        .list();
150
151        for (Map<String, Object> feature : features) {
152
153            Map<String, Object> featureValue =
154                    handle.createQuery("SELECT binary_value, single FROM f_value where id is :fvalue_id").bind("fvalue_id", feature.get("fvalue_id"))
155                            .mapToMap()
156                            .first();
157
158            ContentFeature contentFeature = new ContentFeature();
159            contentFeature.setFeatureType(featureTypeNames.get(feature.get("f_type")).split(":")[0]);
160            contentFeature.setName(featureTypeNames.get(feature.get("f_type")).split(":")[1]);
161            contentFeature.setSingle(Integer.valueOf(1) == featureValue.get("single"));
162            TypeReference<ArrayList<Object>> typeRef
163                    = new TypeReference<>() {
164            };
165
166            try {
167                contentFeature.setValue(OBJECT_MAPPER_MSGPACK.readValue((byte[]) featureValue.get("binary_value"), typeRef));
168            } catch (IOException e) {
169                throw new KodexaException("Unable to unpack value for feature", e);
170            }
171        }
172
173        List<Map<String, Object>> childNodes =
174                handle.createQuery("SELECT id, nt, pid FROM cn where pid is :nodeId").bind("nodeId", contentNode.getUuid())
175                        .mapToMap()
176                        .list();
177
178        for (Map<String, Object> childNode : childNodes) {
179            contentNode.getChildren().add(buildNode(childNode, handle));
180        }
181
182        return contentNode;
183    }
184
185    public byte[] toBytes() {
186        try {
187            return Files.readAllBytes(Path.of(dbPath));
188        } catch (IOException e) {
189            throw new KodexaException("Unable to read KDDB file from " + dbPath);
190        }
191    }
192}
