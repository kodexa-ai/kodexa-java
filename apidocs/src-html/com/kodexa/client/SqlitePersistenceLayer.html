001package com.kodexa.client;
002
003import com.fasterxml.jackson.annotation.JsonInclude;
004import com.fasterxml.jackson.core.JsonProcessingException;
005import com.fasterxml.jackson.core.type.TypeReference;
006import com.fasterxml.jackson.databind.ObjectMapper;
007import org.apache.commons.io.IOUtils;
008import org.jdbi.v3.core.Handle;
009import org.jdbi.v3.core.Jdbi;
010import org.msgpack.jackson.dataformat.MessagePackFactory;
011import org.sqlite.SQLiteConfig;
012
013import java.io.File;
014import java.io.FileOutputStream;
015import java.io.IOException;
016import java.io.InputStream;
017import java.nio.file.Files;
018import java.nio.file.Path;
019import java.util.ArrayList;
020import java.util.HashMap;
021import java.util.List;
022import java.util.Map;
023import java.util.stream.Collectors;
024
025/**
026 * A Persistence Layer that works with the document to allow it to be interacted with
027 */
028public class SqlitePersistenceLayer {
029
030    private final static ObjectMapper OBJECT_MAPPER_MSGPACK;
031    private final Document document;
032
033    Map<Integer, String> nodeTypes;
034
035    Map<Integer, String> featureTypeNames;
036
037    static {
038        OBJECT_MAPPER_MSGPACK = new ObjectMapper(new MessagePackFactory());
039        OBJECT_MAPPER_MSGPACK.setSerializationInclusion(JsonInclude.Include.NON_NULL);
040    }
041
042    private String dbPath;
043    private Jdbi jdbi;
044    private boolean tempFile = false;
045
046    public SqlitePersistenceLayer(Document document) {
047        File file = null;
048        try {
049            this.document = document;
050            file = File.createTempFile("kdx", "kddb");
051            this.dbPath = file.getAbsolutePath();
052            this.tempFile = true;
053            file.deleteOnExit();
054            this.initializeLayer();
055            this.initializeDb();
056        } catch (IOException e) {
057            throw new KodexaException("Unable to initialize the temp file for KDDB", e);
058        }
059    }
060
061    private void initializeDb() {
062
063        jdbi.withHandle(handle -> {
064            handle.execute("CREATE TABLE version (id integer primary key, version text)");
065            handle.execute("CREATE TABLE metadata (id integer primary key, metadata text)");
066            handle.execute("CREATE TABLE cn (id integer primary key, nt INTEGER, pid INTEGER, idx INTEGER)");
067            handle.execute(
068                    "CREATE TABLE cnp (id integer primary key, cn_id INTEGER, pos integer, content text, content_idx integer)");
069
070            handle.execute("CREATE TABLE n_type (id integer primary key, name text)");
071            handle.execute("CREATE TABLE f_type (id integer primary key, name text)");
072            handle.execute(
073                    "CREATE TABLE f_value (id integer primary key, hash integer, binary_value blob, single integer)");
074            handle.execute(
075                    "CREATE TABLE f (id integer primary key, cn_id integer, f_type INTEGER, fvalue_id integer)");
076
077            handle.execute("CREATE UNIQUE INDEX n_type_uk ON n_type(name);");
078            handle.execute("CREATE UNIQUE INDEX f_type_uk ON f_type(name);");
079            handle.execute("CREATE INDEX cn_perf ON cn(nt);");
080            handle.execute("CREATE INDEX cnp_perf ON cnp(cn_id);");
081            handle.execute("CREATE INDEX f_value_hash ON f_value(hash);");
082
083            return handle;
084        });
085    }
086
087    public void close() {
088        if (tempFile) {
089            try {
090                Files.delete(Path.of(this.dbPath));
091            } catch (IOException e) {
092                throw new KodexaException("Unable to delete temp file", e);
093            }
094        }
095    }
096
097    public SqlitePersistenceLayer(InputStream kddbInputStream, Document document) {
098        final File tempFile;
099        try {
100            this.document = document;
101            tempFile = File.createTempFile("kodexa", "kddb");
102            tempFile.deleteOnExit();
103            try (FileOutputStream out = new FileOutputStream(tempFile)) {
104                IOUtils.copy(kddbInputStream, out);
105            }
106            this.dbPath = tempFile.getAbsolutePath();
107
108            this.initializeLayer();
109        } catch (IOException e) {
110            throw new KodexaException("Unable to create persistence layer for KDDB object", e);
111        }
112    }
113
114    public SqlitePersistenceLayer(File kddbFile, Document document) {
115        this.dbPath = kddbFile.getAbsolutePath();
116        this.document = document;
117        this.initializeLayer();
118    }
119
120    private void initializeLayer() {
121        try {
122            Class.forName("org.sqlite.JDBC");
123            SQLiteConfig config = new SQLiteConfig();
124            config.setJournalMode(SQLiteConfig.JournalMode.OFF);
125            jdbi = Jdbi.create("jdbc:sqlite:" + dbPath, config.toProperties());
126        } catch (ClassNotFoundException e) {
127            throw new KodexaException("Unable to create persistence layer for KDDB object", e);
128        }
129    }
130
131    protected void loadDocument() {
132        // This method will update the document to match the contents
133        // of the KDDB database
134        jdbi.withHandle(handle -> {
135
136            // We get all the metadata back
137            try {
138                byte[] metadataPack = (byte[]) handle.createQuery("SELECT * FROM metadata")
139                        .mapToMap()
140                        .first().get("metadata");
141
142                Document baseDocument = OBJECT_MAPPER_MSGPACK.readValue(metadataPack, Document.class);
143                document.setSource(baseDocument.getSource());
144                document.setClasses(baseDocument.getClasses());
145                document.setLabels(baseDocument.getLabels());
146                document.setMetadata(baseDocument.getMetadata());
147                document.setUuid(baseDocument.getUuid());
148
149                // Lets get all the node types and feature type/name combinations
150                nodeTypes =
151                        handle.createQuery("SELECT id, name FROM n_type")
152                                .mapToMap()
153                                .collect(Collectors.toMap(x -> Integer.valueOf(String.valueOf(x.get("id"))), x -> String.valueOf(x.get("name"))));
154
155                featureTypeNames =
156                        handle.createQuery("SELECT id, name FROM f_type")
157                                .mapToMap()
158                                .collect(Collectors.toMap(x -> Integer.valueOf(String.valueOf(x.get("id"))), x -> String.valueOf(x.get("name"))));
159            } catch (IOException e) {
160                throw new KodexaException("Unable to unpack metadata", e);
161            }
162
163            List<Map<String, Object>> contentNodes =
164                    handle.createQuery("SELECT id, nt, pid FROM cn where pid is null")
165                            .mapToMap()
166                            .list();
167            for (Map<String, Object> contentNode : contentNodes) {
168                document.setContentNode(buildNode(contentNode, handle));
169            }
170
171            return contentNodes;
172        });
173    }
174
175    private ContentNode buildNode(Map<String, Object> contentNodeValues, Handle handle) {
176        ContentNode contentNode = new ContentNode();
177        contentNode.setUuid(String.valueOf(contentNodeValues.get("id")));
178
179        List<Map<String, Object>> features =
180                handle.createQuery("SELECT id, f_type, fvalue_id FROM f where cn_id is :nodeId").bind("nodeId", contentNode.getUuid())
181                        .mapToMap()
182                        .list();
183
184        for (Map<String, Object> feature : features) {
185
186            Map<String, Object> featureValue =
187                    handle.createQuery("SELECT binary_value, single FROM f_value where id is :fvalue_id").bind("fvalue_id", feature.get("fvalue_id"))
188                            .mapToMap()
189                            .first();
190
191            ContentFeature contentFeature = new ContentFeature();
192            contentFeature.setFeatureType(featureTypeNames.get(feature.get("f_type")).split(":")[0]);
193            contentFeature.setName(featureTypeNames.get(feature.get("f_type")).split(":")[1]);
194            contentFeature.setSingle(Integer.valueOf(1) == featureValue.get("single"));
195            TypeReference<ArrayList<Object>> typeRef
196                    = new TypeReference<>() {
197            };
198
199            try {
200                contentFeature.setValue(OBJECT_MAPPER_MSGPACK.readValue((byte[]) featureValue.get("binary_value"), typeRef));
201            } catch (IOException e) {
202                throw new KodexaException("Unable to unpack value for feature", e);
203            }
204        }
205
206        List<Map<String, Object>> childNodes =
207                handle.createQuery("SELECT id, nt, pid FROM cn where pid is :nodeId").bind("nodeId", contentNode.getUuid())
208                        .mapToMap()
209                        .list();
210
211        for (Map<String, Object> childNode : childNodes) {
212            contentNode.getChildren().add(buildNode(childNode, handle));
213        }
214
215        return contentNode;
216    }
217
218    public byte[] toBytes() {
219        try {
220            flushMetadata();
221            return Files.readAllBytes(Path.of(dbPath));
222        } catch (IOException e) {
223            throw new KodexaException("Unable to read KDDB file from " + dbPath);
224        }
225    }
226
227    private void flushMetadata() {
228        jdbi.withHandle(handle -> {
229            try {
230                byte[] metadataBytes = OBJECT_MAPPER_MSGPACK.writeValueAsBytes(document);
231                handle.execute("INSERT INTO metadata(metadata, id) VALUES(?, ?)\n" +
232                        "  ON CONFLICT(id) DO UPDATE SET metadata=?", metadataBytes, 1, metadataBytes);
233            } catch (JsonProcessingException e) {
234                throw new KodexaException("Unable to flush metadata to KDDB", e);
235            }
236            return null;
237        });
238
239    }
