001package com.kodexa.client;
002
003import com.fasterxml.jackson.core.JsonParser;
004import com.fasterxml.jackson.core.JsonProcessingException;
005import com.fasterxml.jackson.databind.DeserializationContext;
006import com.fasterxml.jackson.databind.JsonNode;
007import com.fasterxml.jackson.databind.ObjectMapper;
008import com.fasterxml.jackson.databind.deser.std.StdDeserializer;
009import com.fasterxml.jackson.databind.type.CollectionType;
010import com.fasterxml.jackson.databind.type.TypeFactory;
011
012import java.io.IOException;
013import java.util.List;
014
015public class ContentFeatureDeserializer extends StdDeserializer<ContentFeature> {
016
017    private static final ObjectMapper mapper = new ObjectMapper();
018    private static final CollectionType collectionType =
019            TypeFactory
020                    .defaultInstance()
021                    .constructCollectionType(List.class, Object.class);
022
023    public ContentFeatureDeserializer() {
024        this(null);
025    }
026
027    public ContentFeatureDeserializer(Class<ContentFeature> vc) {
028        super(vc);
029    }
030
031    @Override
032    public ContentFeature deserialize(JsonParser jp, DeserializationContext ctxt)
033            throws IOException, JsonProcessingException {
034        JsonNode node = jp.getCodec().readTree(jp);
035        String fullName = node.get("name").asText();
036        Boolean single = node.get("single").asBoolean();
037
038        JsonNode nodeValues = node.get("value");
039
040        ContentFeature contentFeature = new ContentFeature();
041        contentFeature.setFeatureType(fullName.split(":")[0]);
042        contentFeature.setName(fullName.split(":")[1]);
043        contentFeature.setSingle(single);
044
045        if (null != nodeValues
046                && nodeValues.isArray())
047            contentFeature.setValue(mapper.reader(collectionType).readValue(nodeValues));
048
049        return contentFeature;
050    }
051}
