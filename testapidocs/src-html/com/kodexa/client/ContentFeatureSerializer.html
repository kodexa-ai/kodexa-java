001package com.kodexa.client;
002
003import com.fasterxml.jackson.core.JsonGenerator;
004import com.fasterxml.jackson.core.JsonProcessingException;
005import com.fasterxml.jackson.databind.SerializerProvider;
006import com.fasterxml.jackson.databind.ser.std.StdSerializer;
007
008import java.io.IOException;
009
010public class ContentFeatureSerializer extends StdSerializer<ContentFeature> {
011
012    public ContentFeatureSerializer() {
013        this(null);
014    }
015
016    public ContentFeatureSerializer(Class<ContentFeature> t) {
017        super(t);
018    }
019
020    @Override
021    public void serialize(
022            ContentFeature value, JsonGenerator jgen, SerializerProvider provider)
023            throws IOException, JsonProcessingException {
024
025        jgen.writeStartObject();
026        jgen.writeStringField("name", value.getFeatureType() + ":" + value.getName());
027        jgen.writeArrayFieldStart("value");
028        for (Object obj : value.getValue()) {
029            jgen.writeObject(obj);
030        }
031        jgen.writeEndArray();
032        jgen.writeBooleanField("single", value.isSingle());
033        jgen.writeEndObject();
034    }
035}
