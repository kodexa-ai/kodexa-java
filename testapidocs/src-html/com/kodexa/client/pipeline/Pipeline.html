011/**
012 * A Pipeline allows you to put together steps from a connector to a sink
013 * to enable re-use and encapsulation of steps.
014 */
015public class Pipeline {
016    @java.lang.SuppressWarnings("all")
017    private static final org.slf4j.Logger log = org.slf4j.LoggerFactory.getLogger(Pipeline.class);
018    protected final Connector connector;
019    private final PipelineContext context;
020    private Sink sink;
021    private List<PipelineStep> steps = new ArrayList<>();
022
023    public Pipeline(Connector connector) {
024        this.connector = connector;
025        this.context = new PipelineContext();
026    }
027
028    public Pipeline(Document document) {
029        this.connector = new DocumentConnector(document);
030        this.context = new PipelineContext();
031    }
032
033    public void addStep(PipelineStep step) {
034        steps.add(step);
035    }
036
037    public void setSink(Sink sink) {
038        this.sink = sink;
039    }
040
041    public PipelineContext run() {
042        log.info("Starting pipeline");
043        connector.forEachRemaining(document -> {
044            for (PipelineStep step : steps) {
045                log.info("Starting step " + step.getName());
046                long startTime = System.currentTimeMillis();
047                document = step.process(document, context);
048                long endTime = System.currentTimeMillis();
049                log.info("Step processed in " + (endTime - startTime) + " ms");
050            }
051            if (sink != null) {
052                log.info("Writing to sink " + sink.getName());
053                sink.sink(document);
054            }
055        });
056        log.info("Pipeline completed");
057        return context;
058    }
059
060    public static Pipeline fromText(String text) {
061        return new Pipeline(Document.fromText(text));
062    }
063
064    public static Pipeline fromUrl(String url) {
065        return new Pipeline(Document.fromUrl(url));
066    }
