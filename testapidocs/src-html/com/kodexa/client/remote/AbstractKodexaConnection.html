001package com.kodexa.client.remote;
002
003import com.fasterxml.jackson.annotation.JsonInclude;
004import com.fasterxml.jackson.databind.ObjectMapper;
005import org.apache.http.client.config.RequestConfig;
006import org.msgpack.jackson.dataformat.MessagePackFactory;
007
008public abstract class AbstractKodexaConnection {
009
010    protected final static ObjectMapper messagePackOm;
011    protected final static ObjectMapper jsonOm;
012
013    static {
014        messagePackOm = new ObjectMapper(new MessagePackFactory());
015        messagePackOm.setSerializationInclusion(JsonInclude.Include.NON_NULL);
016
017        jsonOm = new ObjectMapper();
018        jsonOm.setSerializationInclusion(JsonInclude.Include.NON_NULL);
019    }
020
021
022    protected RequestConfig getRequestConfig() {
023        int timeout = 120; // seconds
024        return RequestConfig.custom()
025                .setConnectTimeout(timeout * 1000)
026                .setConnectionRequestTimeout(timeout * 1000)
027                .setSocketTimeout(timeout * 1000).build();
028    }
029
030}
