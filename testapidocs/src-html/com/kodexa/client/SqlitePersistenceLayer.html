065    public SqlitePersistenceLayer(Document document) {
066        File file = null;
067        try {
068            this.document = document;
069            file = File.createTempFile("kdx", "kddb");
070            this.dbPath = file.getAbsolutePath();
071            this.tempFile = true;
072            file.deleteOnExit();
073            this.initializeLayer();
074            this.initializeDb();
075
076            md = MessageDigest.getInstance("SHA-1");
077
078        } catch (IOException | NoSuchAlgorithmException e) {
079            throw new KodexaException("Unable to initialize the temp file for KDDB", e);
080        }
081    }

083    private void initializeDb() {
084
085        jdbi.withHandle(handle -> {
086            handle.execute("CREATE TABLE version (id integer primary key, version text)");
087            handle.execute("CREATE TABLE metadata (id integer primary key, metadata text)");
088            handle.execute("CREATE TABLE cn (id integer primary key, nt INTEGER, pid INTEGER, idx INTEGER)");
089            handle.execute(
090                    "CREATE TABLE cnp (id integer primary key, cn_id INTEGER, pos integer, content text, content_idx integer)");
091
092            handle.execute("CREATE TABLE n_type (id integer primary key, name text)");
093            handle.execute("CREATE TABLE f_type (id integer primary key, name text)");
094            handle.execute(
095                    "CREATE TABLE f_value (id integer primary key, hash integer, binary_value blob, single integer)");
096            handle.execute(
097                    "CREATE TABLE f (id integer primary key, cn_id integer, f_type INTEGER, fvalue_id integer)");
098
099            handle.execute("CREATE UNIQUE INDEX n_type_uk ON n_type(name);");
100            handle.execute("CREATE UNIQUE INDEX f_type_uk ON f_type(name);");
101            handle.execute("CREATE INDEX cn_perf ON cn(nt);");
102            handle.execute("CREATE INDEX cnp_perf ON cnp(cn_id);");
103            handle.execute("CREATE INDEX f_value_hash ON f_value(hash);");
104
105            return handle;
106        });
107    }

109    public void close() {
110        if (tempFile) {
111            try {
112                Files.delete(Path.of(this.dbPath));
113            } catch (IOException e) {
114                throw new KodexaException("Unable to delete temp file", e);
115            }
116        }

119    public SqlitePersistenceLayer(InputStream kddbInputStream, Document document) {
120        final File tempFile;
121        try {
122            this.document = document;
123            tempFile = File.createTempFile("kodexa", "kddb");
124            tempFile.deleteOnExit();
125            try (FileOutputStream out = new FileOutputStream(tempFile)) {
126                IOUtils.copy(kddbInputStream, out);
127            }
128            this.dbPath = tempFile.getAbsolutePath();
129
130            this.initializeLayer();
131        } catch (IOException e) {
132            throw new KodexaException("Unable to create persistence layer for KDDB object", e);
133        }
134    }

142    private void initializeLayer() {
143        try {
144            Class.forName("org.sqlite.JDBC");
145            SQLiteConfig config = new SQLiteConfig();
146            config.setJournalMode(SQLiteConfig.JournalMode.OFF);
147            jdbi = Jdbi.create("jdbc:sqlite:" + dbPath, config.toProperties());
148        } catch (ClassNotFoundException e) {
149            throw new KodexaException("Unable to create persistence layer for KDDB object", e);
150        }
151    }

153    protected void loadDocument() {
154        // This method will update the document to match the contents
155        // of the KDDB database
156        jdbi.withHandle(handle -> {
157
158            // We get all the metadata back
159            try {
160                byte[] metadataPack = (byte[]) handle.createQuery("SELECT * FROM metadata")
161                        .mapToMap()
162                        .first().get("metadata");
163
164                Document baseDocument = OBJECT_MAPPER_MSGPACK.readValue(metadataPack, Document.class);
165                document.setSource(baseDocument.getSource());
166                document.setClasses(baseDocument.getClasses());
167                document.setLabels(baseDocument.getLabels());
168                document.setMetadata(baseDocument.getMetadata());
169                document.setUuid(baseDocument.getUuid());
170                document.setMixins(baseDocument.getMixins());
171
172                // Lets get all the node types and feature type/name combinations
173                nodeTypes =
174                        handle.createQuery("SELECT id, name FROM n_type")
175                                .mapToMap()
176                                .collect(Collectors.toMap(x -> Integer.valueOf(String.valueOf(x.get("id"))), x -> String.valueOf(x.get("name"))));
177
178                featureTypeNames =
179                        handle.createQuery("SELECT id, name FROM f_type")
180                                .mapToMap()
181                                .collect(Collectors.toMap(x -> Integer.valueOf(String.valueOf(x.get("id"))), x -> String.valueOf(x.get("name"))));
182            } catch (IOException e) {
183                throw new KodexaException("Unable to unpack metadata", e);
184            }
185
186            List<Map<String, Object>> contentNodes =
187                    handle.createQuery("SELECT id, nt, pid FROM cn where pid is null")
188                            .mapToMap()
189                            .list();
190            for (Map<String, Object> contentNode : contentNodes) {
191                document.setContentNode(buildNode(contentNode, handle));
192            }
193
194            return contentNodes;
195        });
196    }

198    private ContentNode buildNode(Map<String, Object> contentNodeValues, Handle handle) {
199        ContentNode contentNode = new ContentNode();
200        contentNode.setUuid(String.valueOf(contentNodeValues.get("id")));
201        contentNode.setType(nodeTypes.get(contentNodeValues.get("nt")));
202
203        List<Map<String, Object>> features =
204                handle.createQuery("SELECT id, f_type, fvalue_id FROM f where cn_id = :nodeId").bind("nodeId", contentNode.getUuid())
205                        .mapToMap()
206                        .list();
207
208        List<Map<String, Object>> contentParts =
209                handle.createQuery("SELECT content, content_idx FROM cnp where cn_id = :nodeId order by pos asc").bind("nodeId", contentNode.getUuid())
210                        .mapToMap()
211                        .list();
212
213        List<String> parts = new ArrayList<>();
214        contentNode.setContentParts(new ArrayList<>());
215        for (Map<String, Object> contentPart : contentParts) {
216            if (contentPart.get("content") != null) {
217                contentNode.getContentParts().add(contentPart.get("content"));
218                parts.add(String.valueOf(contentPart.get("content")));
219            } else {
220                contentNode.getContentParts().add(contentPart.get("content_idx"));
221            }
222        }
223
224        contentNode.setContent(String.join(" ", parts));
225
226        for (Map<String, Object> feature : features) {
227
228            Map<String, Object> featureValue =
229                    handle.createQuery("SELECT binary_value, single FROM f_value where id = :fvalue_id").bind("fvalue_id", feature.get("fvalue_id"))
230                            .mapToMap()
231                            .first();
232
233            ContentFeature contentFeature = new ContentFeature();
234            contentFeature.setFeatureType(featureTypeNames.get(feature.get("f_type")).split(":")[0]);
235            contentFeature.setName(featureTypeNames.get(feature.get("f_type")).split(":")[1]);
236            contentFeature.setSingle(Integer.valueOf(1) == featureValue.get("single"));
237            TypeReference<ArrayList<Object>> typeRef
238                    = new TypeReference<>() {
239            };
240
241            try {
242                contentFeature.setValue(OBJECT_MAPPER_MSGPACK.readValue((byte[]) featureValue.get("binary_value"), typeRef));
243            } catch (IOException e) {
244                throw new KodexaException("Unable to unpack value for feature", e);
245            }
246
247            contentNode.getFeatures().add(contentFeature);
248        }
249
250        List<Map<String, Object>> childNodes =
251                handle.createQuery("SELECT id, nt, pid FROM cn where pid is :nodeId").bind("nodeId", contentNode.getUuid())
252                        .mapToMap()
253                        .list();
254
255        for (Map<String, Object> childNode : childNodes) {
256            contentNode.getChildren().add(buildNode(childNode, handle));
257        }
258
259        return contentNode;
260    }

262    public byte[] toBytes() {
263        try {
264            flushMetadata();
265            replaceContent();
266            return Files.readAllBytes(Path.of(dbPath));
267        } catch (IOException e) {
268            throw new KodexaException("Unable to read KDDB file from " + dbPath);
269        }
270    }

272    private void replaceContent() {
273        jdbi.withHandle(handle -> {
274            handle.execute("delete from f_value");
275            handle.execute("delete from f");
276            handle.execute("delete from cnp");
277            handle.execute("delete from cn");
278
279            writeNode(handle, document.getContentNode(), null, true);
280
281            return null;
282        });
283
284    }

286    private void writeNode(Handle handle, ContentNode contentNode, Integer parentId, boolean includeChildren) {
287        if (contentNode == null)
288            return;
289
290        int nodeTypeId = getNodeTypeId(handle, contentNode.getType());
291        int nodeId = (int) handle.createUpdate(CONTENT_NODE_INSERT).bind(0, parentId).bind(1, nodeTypeId).bind(2, contentNode.getIndex()).executeAndReturnGeneratedKeys("id").mapToMap().first().get("last_insert_rowid()");
292        contentNode.setUuid(String.valueOf(nodeId));
293
294        if (contentNode.getContentParts() == null || contentNode.getContentParts().isEmpty()) {
295            if (contentNode.getContent() != null) {
296                contentNode.setContentParts(List.of(contentNode.getContent()));
297            }
298        }
299
300        int pos = 0;
301        for (Object contentPart : contentNode.getContentParts()) {
302            if (contentPart instanceof String) {
303                handle.execute(CONTENT_NODE_PART_INSERT, nodeId, pos, contentPart, null);
304            } else {
305                handle.execute(CONTENT_NODE_PART_INSERT, nodeId, pos, null, contentPart);
306            }
307            pos++;
308        }
309
310        for (ContentFeature feature : contentNode.getFeatures()) {
311            writeFeature(handle, feature, contentNode);
312        }
313
314        if (includeChildren) {
315            for (ContentNode child : contentNode.getChildren()) {
316                writeNode(handle, child, nodeId, true);
317            }
318        }
319    }

321    private int writeFeature(Handle handle, ContentFeature feature, ContentNode contentNode) {
322        int fTypeId = getFeatureTypeName(handle, feature.getFeatureType() + ":" + feature.getName());
323
324        // We need to work out the feature value
325        try {
326            byte[] packedValue = OBJECT_MAPPER_MSGPACK.writeValueAsBytes(feature.getValue());
327            Formatter formatter = new Formatter();
328            for (byte b : md.digest(packedValue)) {
329                formatter.format("%02x", b);
330            }
331            long hash = new BigInteger(formatter.toString(), 16).mod(BigDecimal.valueOf(pow(10, 8)).toBigInteger()).longValue();
332            Optional<Map<String, Object>> result = handle.createQuery(FEATURE_VALUE_LOOKUP).bind(0, hash).mapToMap().findFirst();
333            int featureValueId;
334            if (result.isPresent()) {
335                featureValueId = (int) result.get().get("id");
336            } else {
337                featureValueId = (Integer) handle.createUpdate(FEATURE_VALUE_INSERT).bind(0, packedValue).bind(1, hash).bind(2, feature.isSingle()).executeAndReturnGeneratedKeys("id").mapToMap().first().get("last_insert_rowid()");
338            }
339
340            return (Integer) handle.createUpdate(FEATURE_INSERT).bind(0, contentNode.getUuid()).bind(1, fTypeId).bind(2, featureValueId).executeAndReturnGeneratedKeys("id").mapToMap().first().get("last_insert_rowid()");
341
342        } catch (JsonProcessingException e) {
343            throw new KodexaException("Unable to pack feature value", e);
344        }
345
346    }

348    private int getNodeTypeId(Handle handle, String type) {
349        Optional<Map<String, Object>> nodeType = handle.createQuery("SELECT id, name FROM n_type where name is :nodeType").bind("nodeType", type)
350                .mapToMap()
351                .findFirst();
352
353        return nodeType.map(stringObjectMap -> (int) stringObjectMap.get("id")).orElseGet(() -> (Integer) handle.createUpdate("INSERT INTO n_type(name) VALUES(?)").bind(0, type).executeAndReturnGeneratedKeys("id").mapToMap().first().get("last_insert_rowid()"));
354    }
355
356    private int getFeatureTypeName(Handle handle, String type) {
357        Optional<Map<String, Object>> nodeType = handle.createQuery("SELECT id, name FROM f_type where name is :featureType").bind("featureType", type)
358                .mapToMap()
359                .findFirst();
360
361        return nodeType.map(stringObjectMap -> (int) stringObjectMap.get("id")).orElseGet(() -> (Integer) handle.createUpdate("INSERT INTO f_type(name) VALUES(?)").bind(0, type).executeAndReturnGeneratedKeys("id").mapToMap().first().get("last_insert_rowid()"));
362    }

364    private void flushMetadata() {
365        jdbi.withHandle(handle -> {
366            try {
367                Document copyDocument = new Document();
368                copyDocument.setSource(document.getSource());
369                copyDocument.setClasses(document.getClasses());
370                copyDocument.setLabels(document.getLabels());
371                copyDocument.setMetadata(document.getMetadata());
372                copyDocument.setUuid(document.getUuid());
373                copyDocument.setMixins(document.getMixins());
374
375                byte[] metadataBytes = OBJECT_MAPPER_MSGPACK.writeValueAsBytes(copyDocument);
376                handle.execute("INSERT INTO metadata(metadata, id) VALUES(?, ?)\n" +
377                        "  ON CONFLICT(id) DO UPDATE SET metadata=?", metadataBytes, 1, metadataBytes);
378            } catch (JsonProcessingException e) {
379                throw new KodexaException("Unable to flush metadata to KDDB", e);
380            }
381            return null;
382        });
