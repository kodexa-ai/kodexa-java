047    private boolean tempFile = false;

050    public SqlitePersistenceLayer(Document document) {
051        File file = null;
052
053        try {
054            md = MessageDigest.getInstance("SHA-1");
055
056            this.document = document;
057            file = File.createTempFile("kdx", "kddb");
058            this.dbPath = file.getAbsolutePath();
059            this.tempFile = true;
060            file.deleteOnExit();
061            this.initializeLayer();
062            this.initializeDb();
063        } catch (IOException | NoSuchAlgorithmException e) {
064            throw new KodexaException("Unable to initialize the temp file for KDDB", e);
065        }
066    }

068    private void initializeDb() {
069
070        jdbi.withHandle(handle -> {
071            handle.execute("CREATE TABLE version (id integer primary key, version text)");
072            handle.execute("CREATE TABLE metadata (id integer primary key, metadata text)");
073            handle.execute("CREATE TABLE cn (id integer primary key, nt INTEGER, pid INTEGER, idx INTEGER)");
074            handle.execute(
075                    "CREATE TABLE cnp (id integer primary key, cn_id INTEGER, pos integer, content text, content_idx integer)");
076
077            handle.execute("CREATE TABLE n_type (id integer primary key, name text)");
078            handle.execute("CREATE TABLE f_type (id integer primary key, name text)");
079
080            handle.execute(
081                    "CREATE TABLE ft (id integer primary key, cn_id integer, f_type INTEGER, binary_value blob, single integer, tag_uuid text)");
082
083            handle.execute("CREATE UNIQUE INDEX n_type_uk ON n_type(name);");
084            handle.execute("CREATE UNIQUE INDEX f_type_uk ON f_type(name);");
085            handle.execute("CREATE INDEX cn_perf ON cn(nt);");
086            handle.execute("CREATE INDEX cn_perf2 ON cn(pid);");
087            handle.execute("CREATE INDEX cnp_perf ON cnp(cn_id, pos);");
088            handle.execute("CREATE INDEX f_perf ON ft(cn_id);");
089            handle.execute("CREATE INDEX f_perf2 ON ft(tag_uuid);");
090
091            return handle;
092        });
093    }

095    public void close() {
096        if (tempFile) {
097            try {
098                Files.delete(Path.of(this.dbPath));
099            } catch (IOException e) {
100                throw new KodexaException("Unable to delete temp file", e);
101            }
102        }

105    public SqlitePersistenceLayer(InputStream kddbInputStream, Document document) {
106        final File tempFile;
107        try {
108            md = MessageDigest.getInstance("SHA-1");
109
110            this.document = document;
111            tempFile = File.createTempFile("kodexa", "kddb");
112            tempFile.deleteOnExit();
113            try (FileOutputStream out = new FileOutputStream(tempFile)) {
114                IOUtils.copy(kddbInputStream, out);
115            }
116            this.dbPath = tempFile.getAbsolutePath();
117            this.initializeLayer();
118            this.loadDocument();
119        } catch (IOException | NoSuchAlgorithmException e) {
120            throw new KodexaException("Unable to create persistence layer for KDDB object", e);
121        }
122    }

138    private void initializeLayer() {
139        try {
140            Class.forName("org.sqlite.JDBC");
141            SQLiteConfig config = new SQLiteConfig();
142            config.setJournalMode(SQLiteConfig.JournalMode.OFF);
143            jdbi = Jdbi.create("jdbc:sqlite:" + dbPath, config.toProperties());
144        } catch (ClassNotFoundException e) {
145            throw new KodexaException("Unable to create persistence layer for KDDB object", e);
146        }
147    }

149    protected void loadDocument() {
150        // This method will update the document to match the contents
151        // of the KDDB database
152        jdbi.withHandle(handle -> {
153
154            // We get all the metadata back
155            try {
156                byte[] metadataPack = (byte[]) handle.createQuery("SELECT * FROM metadata")
157                        .mapToMap()
158                        .first().get("metadata");
159
160                Document baseDocument = OBJECT_MAPPER_MSGPACK.readValue(metadataPack, Document.class);
161                document.setSource(baseDocument.getSource());
162                document.setClasses(baseDocument.getClasses());
163                document.setLabels(baseDocument.getLabels());
164                document.setMetadata(baseDocument.getMetadata());
165                document.setUuid(baseDocument.getUuid());
166                document.setMixins(baseDocument.getMixins());
167                document.setVersion(baseDocument.getVersion());
168
169
170                if (document.getVersion().equals("4.0.0") || document.getVersion().equals("2.0.0")) {
171
172                    handle.execute("CREATE TABLE ft (id integer primary key,cn_id integer,f_type INTEGER, binary_value blob,single integer,tag_uuid text)");
173                    handle.execute("insert into ft select f.id, f.cn_id, f.f_type, fv.binary_value, fv.single, null from f, f_value fv where fv.id = f.id");
174                    handle.execute("drop table f");
175                    handle.execute("drop table f_value");
176                    handle.execute("CREATE INDEX f_perf ON ft(cn_id);");
177                    handle.execute("CREATE INDEX f_perf2 ON ft(tag_uuid);");
178                    document.setVersion("4.0.1");
179                    flushMetadata();
180                }
181
182                // Lets get all the node types and feature type/name combinations
183                nodeTypes =
184                        handle.createQuery("SELECT id, name FROM n_type")
185                                .mapToMap()
186                                .collect(Collectors.toMap(x -> Integer.valueOf(String.valueOf(x.get("id"))), x -> String.valueOf(x.get("name"))));
187
188                featureTypeNames =
189                        handle.createQuery("SELECT id, name FROM f_type")
190                                .mapToMap()
191                                .collect(Collectors.toMap(x -> Integer.valueOf(String.valueOf(x.get("id"))), x -> String.valueOf(x.get("name"))));
192            } catch (IOException e) {
193                throw new KodexaException("Unable to unpack metadata", e);
194            }
195
196            List<Map<String, Object>> contentNodes =
197                    handle.createQuery("SELECT id, nt, pid FROM cn where pid is null")
198                            .mapToMap()
199                            .list();
200            for (Map<String, Object> contentNode : contentNodes) {
201                document.setContentNode(buildNode(contentNode, handle));
202            }
203
204            return contentNodes;
205        });
206    }

208    public void updateNode(ContentNode node) {

211            return null;

215    private ContentNode buildNode(Map<String, Object> contentNodeValues, Handle handle) {
216        ContentNode contentNode = new ContentNode(this.document);
217        contentNode.setUuid(String.valueOf(contentNodeValues.get("id")));
218        contentNode.setType(nodeTypes.get(contentNodeValues.get("nt")));
219
220        List<Map<String, Object>> features =
221                handle.createQuery("SELECT id, f_type, binary_value, single FROM ft where cn_id = :nodeId").bind("nodeId", contentNode.getUuid())
222                        .mapToMap()
223                        .list();
224
225        List<Map<String, Object>> contentParts =
226                handle.createQuery("SELECT content, content_idx FROM cnp where cn_id = :nodeId order by pos asc").bind("nodeId", contentNode.getUuid())
227                        .mapToMap()
228                        .list();
229
230        List<String> parts = new ArrayList<>();
231        contentNode.setContentParts(new ArrayList<>());
232        for (Map<String, Object> contentPart : contentParts) {
233            if (contentPart.get("content") != null) {
234                contentNode.getContentParts().add(contentPart.get("content"));
235                parts.add(String.valueOf(contentPart.get("content")));
236            } else {
237                contentNode.getContentParts().add(contentPart.get("content_idx"));
238            }
239        }
240
241        contentNode.setContent(String.join(" ", parts));
242
243        for (Map<String, Object> feature : features) {
244            ContentFeature contentFeature = new ContentFeature();
245            contentFeature.setFeatureType(featureTypeNames.get(feature.get("f_type")).split(":")[0]);
246            contentFeature.setName(featureTypeNames.get(feature.get("f_type")).split(":")[1]);
247            contentFeature.setSingle(Integer.valueOf(1) == feature.get("single"));
248            TypeReference<ArrayList<Object>> typeRef
249                    = new TypeReference<>() {
250            };
251
252            try {
253                contentFeature.setValue(OBJECT_MAPPER_MSGPACK.readValue((byte[]) feature.get("binary_value"), typeRef));
254            } catch (IOException e) {
255                throw new KodexaException("Unable to unpack value for feature", e);
256            }
257
258            contentNode.getFeatures().add(contentFeature);
259        }
260
261        return contentNode;
262    }

273    private void updateNode(Handle handle, ContentNode contentNode, Integer parentId, boolean includeChildren) {
274        if (contentNode == null)
275            return;
276
277        int nodeTypeId = getNodeTypeId(handle, contentNode.getType());
278
279        if (contentNode.getUuid() == null) {
280            int nodeId = (int) handle.createUpdate(CONTENT_NODE_INSERT).bind(0, parentId).bind(1, nodeTypeId).bind(2, contentNode.getIndex()).executeAndReturnGeneratedKeys("id").mapToMap().first().get("last_insert_rowid()");
281            contentNode.setUuid(String.valueOf(nodeId));
282        } else {
283            // We need to delete everything for the node to replace it
284            handle.execute("delete from ft where cn_id=?", contentNode.getUuid());
285            handle.execute("delete from cnp where cn_id=?", contentNode.getUuid());
286            handle.execute("delete from cn  where id=?", contentNode.getUuid());
287
288            handle.createUpdate(CONTENT_NODE_INSERT_WITH_ID)
289                    .bind(0, contentNode.getUuid())
290                    .bind(1, parentId)
291                    .bind(2, nodeTypeId)
292                    .bind(3, contentNode.getIndex())
293                    .execute();
294        }
295
296        if (contentNode.getContentParts() == null || contentNode.getContentParts().isEmpty()) {
297            if (contentNode.getContent() != null) {
298                contentNode.setContentParts(List.of(contentNode.getContent()));
299            }
300        }
301
302        int pos = 0;
303        for (Object contentPart : contentNode.getContentParts()) {
304            if (contentPart instanceof String) {
305                handle.execute(CONTENT_NODE_PART_INSERT, contentNode.getUuid(), pos, contentPart, null);
306            } else {
307                handle.execute(CONTENT_NODE_PART_INSERT, contentNode.getUuid(), pos, null, contentPart);
308            }
309            pos++;
310        }
311
312        for (ContentFeature feature : contentNode.getFeatures()) {
313            writeFeature(handle, feature, contentNode);
314        }
315
316        if (includeChildren) {
317            for (ContentNode child : contentNode.getChildren()) {
318                updateNode(handle, child, Integer.valueOf(contentNode.getUuid()), true);
319            }
320        }
321    }

323    private int writeFeature(Handle handle, ContentFeature feature, ContentNode contentNode) {
324        int fTypeId = getFeatureTypeName(handle, feature.getFeatureType() + ":" + feature.getName());
325
326        // We need to work out the feature value
327        try {
328            byte[] packedValue = OBJECT_MAPPER_MSGPACK.writeValueAsBytes(feature.getValue());
329            Formatter formatter = new Formatter();
330            for (byte b : md.digest(packedValue)) {
331                formatter.format("%02x", b);
332            }
333            String tagUuid = null;
334
335            if ("tag".equals(feature.getFeatureType()) && feature.getValue().size() > 0) {
336                tagUuid = (String) ((Map) feature.getValue().get(0)).get("uuid");
337            }
338
339            return (Integer) handle.createUpdate(FEATURE_INSERT).bind(0, contentNode.getUuid()).bind(1, fTypeId).bind(2, packedValue).bind(3, feature.isSingle()).bind(4, tagUuid).executeAndReturnGeneratedKeys("id").mapToMap().first().get("last_insert_rowid()");
340
341        } catch (JsonProcessingException e) {
342            throw new KodexaException("Unable to pack feature value", e);
343        }
344
345    }

347    private int getNodeTypeId(Handle handle, String type) {
348        Optional<Map<String, Object>> nodeType = handle.createQuery("SELECT id, name FROM n_type where name is :nodeType").bind("nodeType", type)
349                .mapToMap()
350                .findFirst();
351
352        return nodeType.map(stringObjectMap -> (int) stringObjectMap.get("id")).orElseGet(() -> (Integer) handle.createUpdate("INSERT INTO n_type(name) VALUES(?)").bind(0, type).executeAndReturnGeneratedKeys("id").mapToMap().first().get("last_insert_rowid()"));
353    }

355    private int getFeatureTypeName(Handle handle, String type) {
356        Optional<Map<String, Object>> nodeType = handle.createQuery("SELECT id, name FROM f_type where name is :featureType").bind("featureType", type)
357                .mapToMap()
358                .findFirst();
359
360        return nodeType.map(stringObjectMap -> (int) stringObjectMap.get("id")).orElseGet(() -> (Integer) handle.createUpdate("INSERT INTO f_type(name) VALUES(?)").bind(0, type).executeAndReturnGeneratedKeys("id").mapToMap().first().get("last_insert_rowid()"));
361    }

363    private void flushMetadata() {
364        jdbi.withHandle(handle -> {
365            try {
366                Document copyDocument = new Document();
367                copyDocument.setUuid(document.getUuid());
368                copyDocument.setSource(document.getSource());
369                copyDocument.setClasses(document.getClasses());
370                copyDocument.setLabels(document.getLabels());
371                copyDocument.setMetadata(document.getMetadata());
372                copyDocument.setUuid(document.getUuid());
373                copyDocument.setMixins(document.getMixins());
374                copyDocument.setVersion(document.getVersion());
375
376                byte[] metadataBytes = OBJECT_MAPPER_MSGPACK.writeValueAsBytes(copyDocument);
377                handle.execute("INSERT INTO metadata(metadata, id) VALUES(?, ?)\n" +
378                        "  ON CONFLICT(id) DO UPDATE SET metadata=?", metadataBytes, 1, metadataBytes);
379            } catch (JsonProcessingException e) {
380                throw new KodexaException("Unable to flush metadata to KDDB", e);
381            }
382            return null;
383        });
384    }

386    public List<ContentNode> getChildNodes(ContentNode contentNode) {
387        return jdbi.withHandle(handle -> {
388            List<Map<String, Object>> childNodes =
389                    handle.createQuery("SELECT id, nt, pid FROM cn where pid is :nodeId").bind("nodeId", contentNode.getUuid())
390                            .mapToMap()
391                            .list();
392            List<ContentNode> children = new ArrayList<>();
393            for (Map<String, Object> childNode : childNodes) {
394                ContentNode child = buildNode(childNode, handle);
395                child.setParent(contentNode);
396                children.add(child);
397            }
398
399            return children;
400        });
401    }

407    public int getNodeCountByType(String type) {
408        return (int) jdbi.withHandle(handle -> handle.createQuery("select count(1) as num from cn where nt=:nt").bind("nt", nodeTypes.entrySet()
409                .stream()
410                .filter(entry -> type.equals(entry.getValue()))
411                .map(Map.Entry::getKey).findFirst().orElse(-1)).mapToMap().first().get("num"));

423    public List<ContentNode> getTaggedNodes() {
424
425        // Get all the feature types that are tags - then lets find all those nodes
426        return jdbi.withHandle(handle -> {
427            List<ContentNode> nodes = new ArrayList<>();
428            List<Map<String, Object>> contentNodesRaw =
429                    handle.createQuery("select * from cn where id in (select cn_id from ft where f_type in (select id from f_type where name like 'tag:%'))")
430                            .mapToMap()
431                            .list();
432            for (Map<String, Object> contentNodeRaw : contentNodesRaw) {
433                nodes.add(buildNode(contentNodeRaw, handle));
434            }
435            return nodes;
436        });
437    }
