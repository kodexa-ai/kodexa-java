001package com.kodexa.client;
002
003import com.fasterxml.jackson.annotation.JsonInclude;
004import com.fasterxml.jackson.core.JsonProcessingException;
005import com.fasterxml.jackson.core.type.TypeReference;
006import com.fasterxml.jackson.databind.ObjectMapper;
007import org.apache.commons.io.IOUtils;
008import org.jdbi.v3.core.Handle;
009import org.jdbi.v3.core.Jdbi;
010import org.msgpack.jackson.dataformat.MessagePackFactory;
011
012import java.io.File;
013import java.io.FileOutputStream;
014import java.io.IOException;
015import java.io.InputStream;
016import java.nio.file.Files;
017import java.nio.file.Path;
018import java.util.ArrayList;
019import java.util.HashMap;
020import java.util.List;
021import java.util.Map;
022import java.util.stream.Collectors;
023
024/**
025 * A Persistence Layer that works with the document to allow it to be interacted with
026 */
027public class SqlitePersistenceLayer {
028
029    private final static ObjectMapper OBJECT_MAPPER_MSGPACK;
030    private final Document document;
031
032    Map<Integer, String> nodeTypes;
033
034    Map<Integer, String> featureTypeNames;
035
036    static {
037        OBJECT_MAPPER_MSGPACK = new ObjectMapper(new MessagePackFactory());
038        OBJECT_MAPPER_MSGPACK.setSerializationInclusion(JsonInclude.Include.NON_NULL);
039    }
040
041    private String dbPath;
042    private Jdbi jdbi;
043    private boolean tempFile = false;
044
045    public SqlitePersistenceLayer(Document document) {
046        File file = null;
047        try {
048            this.document = document;
049            file = File.createTempFile("kdx", "kddb");
050            this.dbPath = file.getAbsolutePath();
051            this.tempFile = true;
052            file.deleteOnExit();
053            this.initializeLayer();
054            this.initializeDb();
055        } catch (IOException e) {
056            throw new KodexaException("Unable to initialize the temp file for KDDB", e);
057        }
058    }
059
060    private void initializeDb() {
061
062        jdbi.withHandle(handle -> {
063            handle.execute("CREATE TABLE version (id integer primary key, version text)");
064            handle.execute("CREATE TABLE metadata (id integer primary key, metadata text)");
065            handle.execute("CREATE TABLE cn (id integer primary key, nt INTEGER, pid INTEGER, idx INTEGER)");
066            handle.execute(
067                    "CREATE TABLE cnp (id integer primary key, cn_id INTEGER, pos integer, content text, content_idx integer)");
068
069            handle.execute("CREATE TABLE n_type (id integer primary key, name text)");
070            handle.execute("CREATE TABLE f_type (id integer primary key, name text)");
071            handle.execute(
072                    "CREATE TABLE f_value (id integer primary key, hash integer, binary_value blob, single integer)");
073            handle.execute(
074                    "CREATE TABLE f (id integer primary key, cn_id integer, f_type INTEGER, fvalue_id integer)");
075
076            handle.execute("CREATE UNIQUE INDEX n_type_uk ON n_type(name);");
077            handle.execute("CREATE UNIQUE INDEX f_type_uk ON f_type(name);");
078            handle.execute("CREATE INDEX cn_perf ON cn(nt);");
079            handle.execute("CREATE INDEX cnp_perf ON cnp(cn_id);");
080            handle.execute("CREATE INDEX f_value_hash ON f_value(hash);");
081            return null;
082        });
083    }
084
085    public void close() {
086        if (tempFile) {
087            try {
088                Files.delete(Path.of(this.dbPath));
089            } catch (IOException e) {
090                throw new KodexaException("Unable to delete temp file", e);
091            }
092        }
093    }
094
095    public SqlitePersistenceLayer(InputStream kddbInputStream, Document document) {
096        final File tempFile;
097        try {
098            this.document = document;
099            tempFile = File.createTempFile("kodexa", "kddb");
100            tempFile.deleteOnExit();
101            try (FileOutputStream out = new FileOutputStream(tempFile)) {
102                IOUtils.copy(kddbInputStream, out);
103            }
104            this.dbPath = tempFile.getAbsolutePath();
105
106            this.initializeLayer();
107        } catch (IOException e) {
108            throw new KodexaException("Unable to create persistence layer for KDDB object", e);
109        }
110    }
111
112    public SqlitePersistenceLayer(File kddbFile, Document document) {
113        this.dbPath = kddbFile.getAbsolutePath();
114        this.document = document;
115        this.initializeLayer();
116    }
117
118    private void initializeLayer() {
119        try {
120            Class.forName("org.sqlite.JDBC");
121            jdbi = Jdbi.create("jdbc:sqlite:" + dbPath);
122        } catch (ClassNotFoundException e) {
123            throw new KodexaException("Unable to create persistence layer for KDDB object", e);
124        }
125    }
126
127    protected void loadDocument() {
128        // This method will update the document to match the contents
129        // of the KDDB database
130        jdbi.withHandle(handle -> {
131
132            // We get all the metadata back
133            try {
134                byte[] metadataPack = (byte[]) handle.createQuery("SELECT * FROM metadata")
135                        .mapToMap()
136                        .first().get("metadata");
137
138                Document baseDocument = OBJECT_MAPPER_MSGPACK.readValue(metadataPack, Document.class);
139                document.setSource(baseDocument.getSource());
140                document.setClasses(baseDocument.getClasses());
141                document.setLabels(baseDocument.getLabels());
142                document.setMetadata(baseDocument.getMetadata());
143                document.setUuid(baseDocument.getUuid());
144
145                // Lets get all the node types and feature type/name combinations
146                nodeTypes =
147                        handle.createQuery("SELECT id, name FROM n_type")
148                                .mapToMap()
149                                .collect(Collectors.toMap(x -> Integer.valueOf(String.valueOf(x.get("id"))), x -> String.valueOf(x.get("name"))));
150
151                featureTypeNames =
152                        handle.createQuery("SELECT id, name FROM f_type")
153                                .mapToMap()
154                                .collect(Collectors.toMap(x -> Integer.valueOf(String.valueOf(x.get("id"))), x -> String.valueOf(x.get("name"))));
155            } catch (IOException e) {
156                throw new KodexaException("Unable to unpack metadata", e);
157            }
158
159            List<Map<String, Object>> contentNodes =
160                    handle.createQuery("SELECT id, nt, pid FROM cn where pid is null")
161                            .mapToMap()
162                            .list();
163            for (Map<String, Object> contentNode : contentNodes) {
164                document.setContentNode(buildNode(contentNode, handle));
165            }
166
167            return contentNodes;
168        });
169    }
170
171    private ContentNode buildNode(Map<String, Object> contentNodeValues, Handle handle) {
172        ContentNode contentNode = new ContentNode();
173        contentNode.setUuid(String.valueOf(contentNodeValues.get("id")));
174
175        List<Map<String, Object>> features =
176                handle.createQuery("SELECT id, f_type, fvalue_id FROM f where cn_id is :nodeId").bind("nodeId", contentNode.getUuid())
177                        .mapToMap()
178                        .list();
179
180        for (Map<String, Object> feature : features) {
181
182            Map<String, Object> featureValue =
183                    handle.createQuery("SELECT binary_value, single FROM f_value where id is :fvalue_id").bind("fvalue_id", feature.get("fvalue_id"))
184                            .mapToMap()
185                            .first();
186
187            ContentFeature contentFeature = new ContentFeature();
188            contentFeature.setFeatureType(featureTypeNames.get(feature.get("f_type")).split(":")[0]);
189            contentFeature.setName(featureTypeNames.get(feature.get("f_type")).split(":")[1]);
190            contentFeature.setSingle(Integer.valueOf(1) == featureValue.get("single"));
191            TypeReference<ArrayList<Object>> typeRef
192                    = new TypeReference<>() {
193            };
194
195            try {
196                contentFeature.setValue(OBJECT_MAPPER_MSGPACK.readValue((byte[]) featureValue.get("binary_value"), typeRef));
197            } catch (IOException e) {
198                throw new KodexaException("Unable to unpack value for feature", e);
199            }
200        }
201
202        List<Map<String, Object>> childNodes =
203                handle.createQuery("SELECT id, nt, pid FROM cn where pid is :nodeId").bind("nodeId", contentNode.getUuid())
204                        .mapToMap()
205                        .list();
206
207        for (Map<String, Object> childNode : childNodes) {
208            contentNode.getChildren().add(buildNode(childNode, handle));
209        }
210
211        return contentNode;
212    }
213
214    public byte[] toBytes() {
215        try {
216            flushMetadata();
217            return Files.readAllBytes(Path.of(dbPath));
218        } catch (IOException e) {
219            throw new KodexaException("Unable to read KDDB file from " + dbPath);
220        }
221    }
222
223    private void flushMetadata() {
224        jdbi.withHandle(handle -> {
225            try {
226                handle.execute("update metadata set metadata=? where id=1", OBJECT_MAPPER_MSGPACK.writeValueAsBytes(document));
227            } catch (JsonProcessingException e) {
228                throw new KodexaException("Unable to flush metadata to KDDB", e);
229            }
230            return null;
231        });
232
233    }
