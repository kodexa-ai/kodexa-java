9   public class ParameterizedPipelineTest {
10  
11      @Test
12      public void basicParameterTest() {
13          Pipeline pipeline = Pipeline.fromText("Hello World");
14          pipeline.addParameter("burger", "bar");
15          pipeline.addStep(MetadataSetStep.class, Options.start().set("key", "cheese").set("value", "${burger}").parameterized(true));
16          PipelineContext context = pipeline.run();
17          Assert.assertTrue(context.getOutputDocument().getMetadata().containsKey("cheese"));
18          Assert.assertSame("bar", context.getOutputDocument().getMetadata().get("cheese"));
19      }
