3   import com.kodexa.client.pipeline.OptionDrivenStep;
4   import com.kodexa.client.pipeline.Options;
5   import com.kodexa.client.pipeline.PipelineContext;
6   import com.kodexa.client.steps.PipelineStep;
7   
8   public class MetadataSetStep implements PipelineStep, OptionDrivenStep {
9   
10      private Options options;
11  
12      @Override
13      public Document process(Document document, PipelineContext context) {
14  
15          if (options.get().containsKey("key")) {
16              document.getMetadata().put(String.valueOf(options.get().get("key")), options.get().get("value"));
17          }
18  
19          return document;
20      }
21  
22      @Override
23      public String getName() {
24          return "Metadata Set Step";
25      }
26  
27      @Override
28      public void setOptions(Options options) {
29          this.options = options;
30      }
31  }
